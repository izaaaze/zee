    <script>
        function startCamera() {
            // Menyembunyikan tombol setelah diklik
            document.querySelector('.button').style.display = 'none';

            // Menampilkan video kamera
            document.getElementById('cameraContainer').style.display = 'block';

            // Mengakses kamera
            navigator.mediaDevices.getUserMedia({ video: true })
                .then(function(stream) {
                    const videoElement = document.getElementById('video');
                    videoElement.srcObject = stream;
                })
                .catch(function(error) {
                    console.error("Tidak dapat mengakses kamera: ", error);
                });
        }

        function toggleCoding() {
            var codingDisplay = document.getElementById('codingDisplay');
            if (codingDisplay.style.display === 'none') {
                codingDisplay.style.display = 'block';
            } else {
                codingDisplay.style.display = 'none';
            }
        }
